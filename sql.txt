 CREATE TABLE users (ID MEDIUMINT NOT NULL AUTO_INCREMENT PRIMARY KEY,
username VARCHAR(60), email VARCHAR(60), name VARCHAR(60), poweruser
BOOLEAN, salt VARCHAR(10), password VARCHAR(60)) ;


 CREATE TABLE offerings (ID MEDIUMINT NOT NULL AUTO_INCREMENT PRIMARY KEY,
user_id MEDIUMINT, name VARCHAR(60), email VARCHAR(60), type VARCHAR(10), var VARCHAR(50), desc TEXT) ;


 CREATE TABLE blogs (ID MEDIUMINT NOT NULL AUTO_INCREMENT PRIMARY KEY,
user_id MEDIUMINT, title VARCHAR(60), story TEXT) ;


 CREATE TABLE sugs (ID MEDIUMINT NOT NULL AUTO_INCREMENT PRIMARY KEY,
user_id MEDIUMINT, email VARCHAR(60), name VARCHAR(60), message TEXT) ;
